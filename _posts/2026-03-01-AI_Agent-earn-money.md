#### 🔧 技术实现代码示例

```python
# 跨境电商客服Agent核心代码
# 使用OpenClaw本地部署

from easyclaw import Agent, Workflow
from ecommerce_apis import ShopeeAPI, LazadaAPI

class CrossBorderCustomerServiceAgent:
    def __init__(self):
        self.agent = Agent(model="claude.sonnet-4.6")
        self.knowledge_base = self.load_product_kb()

    async def handle_inquiry(self, message, platform, language):
        # 意图识别
        intent = await self.agent.classify_intent(message)

        if intent == "order_status":
            order_id = self.extract_order_id(message)
            status = await self.query_order(order_id)
            response = await self.agent.generate_response(
                intent=intent,
                data=status,
                language=language,
                tone="friendly"
            )

        elif intent == "product_inquiry":
            product_info = self.search_kb(message)
            response = await self.agent.generate_response(
                intent=intent,
                data=product_info,
                language=language
            )

        elif intent == "return_request":
            response = await self.handle_return(message, language)

        else:
            # 复杂问题转人工，同时生成建议回复
            response = await self.escalate_with_suggestion(message)

        return response
```

# 使用EasyClaw的财务发票自动化Agent
# 文件: invoice_automation_agent.py

from easyclaw import Agent, DesktopControl
import pandas as pd

class InvoiceProcessingAgent:
    def __init__(self):
        self.agent = Agent(model="claude.sonnet-4.6")
        self.desktop = DesktopControl()

    def process_invoice_folder(self, folder_path):
        """处理整个文件夹的发票"""
        invoices = self.desktop.get_files(folder_path, ".pdf,.jpg,.png")

        results = []
        for invoice in invoices:
            data = self.extract_invoice_data(invoice)
            results.append(data)

        # 生成Excel报告
        df = pd.DataFrame(results)
        df.to_excel(f"发票汇总_{datetime.now().date()}.xlsx")

        return len(results)

    def extract_invoice_data(self, file_path):
        """使用AI提取发票信息"""
        # 截图或读取PDF
        image = self.desktop.capture(file_path)

        # AI识别
        prompt = """
        请从这张发票图片中提取以下信息，以JSON格式返回：
        {
            "invoice_number": "发票号码",
            "date": "开票日期",
            "seller": "销售方",
            "buyer": "购买方",
            "amount": "金额",
            "tax": "税额",
            "total": "价税合计",
            "items": ["商品列表"]
        }
        """

        result = self.agent.vision_analyze(image, prompt)
        return json.loads(result)
